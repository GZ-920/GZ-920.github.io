# 课程源码
## 源码
```cc
#include<iostream>
#include<string>
using namespace std;
int main() {
	int a;
	cout << "请输入一个整型：";
	cin >> a;
	cout << "你输入的整型是：" << a << endl;
	float b;
	cout << "请输入一个浮点型：";
	cin >> b;
	cout << "你输入的浮点型是：" << b << endl;
	string c;
	cout << "请输入一个字符型：";
	cin >> c;
	cout << "你输入的字符型是：" << c << endl;
	return 0;
}
```
